// --- カメラ追従ロジック (update関数内で毎フレーム実行) --- function updateCamera() { // 戦車の背後 10ユニット、高さ 5ユニットの位置をターゲットにする const relativeCameraOffset = new THREE.Vector3(0, 5, -10); // 戦車の現在の向きに合わせてオフセットを回転 const cameraOffset = relativeCameraOffset.applyMatrix4(tank.matrixWorld); // カメラの位置を滑らかに更新 (Lerpでスムーズな動きに) camera.position.lerp(cameraOffset, 0.1); // 常に戦車の少し先（照準方向）を見つめるように設定 const lookAtPos = tank.position.clone().add(new THREE.Vector3(0, 2, 0)); camera.lookAt(lookAtPos); } // --- セカンダリ武装：火炎放射器 (Flamethrower) の実装 --- class Flamethrower { constructor(scene, physicsWorld) { this.scene = scene; this.world = physicsWorld; this.isFiring = false; this.range = 8; this.lastFireTime = 0; this.fireRate = 50; // 連射速度（ミリ秒） } fire(turret) { const now = Date.now(); if (now - this.lastFireTime < this.fireRate) return; this.lastFireTime = now; // 砲塔の先端から炎のパーティクルを生成 const fireParticle = new THREE.Mesh( new THREE.SphereGeometry(0.2), new THREE.MeshBasicMaterial({ color: 0xff4500, transparent: true, opacity: 0.8 }) ); // 砲塔の向きに合わせて初速と位置を設定 const direction = new THREE.Vector3(0, 0, 1).applyQuaternion(turret.quaternion); fireParticle.position.copy(turret.position).add(direction.clone().multiplyScalar(2)); // わずかにランダムな広がり（扇形）を追加 direction.x += (Math.random() - 0.5) * 0.2; direction.y += (Math.random() - 0.5) * 0.2; this.scene.add(fireParticle); // 物理演算（Matter.js等）での判定用ボディ生成 const body = Matter.Bodies.circle(fireParticle.position.x, fireParticle.position.z, 0.5, { isSensor: true, // 障害物をすり抜けるが判定は持つ設定 label: "flame" }); Matter.Body.setVelocity(body, { x: direction.x * 5, y: direction.z * 5 }); Matter.World.add(this.world, body); // 一定時間で消滅（射程制限） setTimeout(() => { this.scene.remove(fireParticle); Matter.World.remove(this.world, body); }, 600); } } // --- メインループ内 --- function animate() { requestAnimationFrame(animate); // 戦車の移動入力処理 handleTankMovement(); // カメラを戦車に追従させる updateCamera(); // セカンダリ武装 [Qキー] の判定 if (keys['q'] || isMobileQButtonPressed) { if (currentSecondary === 'Flamethrower') { flamethrower.fire(turret); } } renderer.render(scene, camera); }
